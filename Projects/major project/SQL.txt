1. SELECT YEAR(STR_TO_DATE(ORDERDATE,'%d-%m-%Y')) AS sales_year, SUM(SALES)
   AS total_sales
   FROM sales2
   GROUP BY
   YEAR(STR_TO_DATE(ORDERDATE,'%d-%m-%Y'))
   ORDER BY sales_year ASC;

2. SELECT PRODUCTLINE
   FROM orders
   WHERE YEAR(order_date)='2004'
   GROUP BY PRODUCTLINE
   HAVING COUNT(*)>40;

3. SELECT * FROM sales2
   WHERE SALES=(SELECT MAX(SALES) FROM sales2;

4. SELECT PRODUCTLINE, AVG(PRICEEACH) AS average_price
   FROM sales2
   GROUP BY PRODUCTLINE;
   
5. SELECT QTR_ID, COUNT(*) AS ORDDER_COUNT
   FROM sales2
   WHERE PRODUCTLINE='Motorcycles'
   GROUP BY QTR_ID;

6. SELECT ORDERDATE
   FROM sales2
   WHERE STATUS='Shipped' AND YEAR(ORDERDATE)=2004;

7. SELECT PRODUCTCODE, SUM(SALES) AS total_sales
   FROM orders
   GROUP BY PRODUCTCODE
   ORDER BY total_sales DESC;

8. SELECT * FROM orders
   WHERE PRICEEACH=(SELECT MIN(PRICEEACH) FROM orders);

9. SELECT YEAR(ORDERDATE) AS YEAR, MONTH(ORDERDATE) AS MONTH, SUM(SALES) AS TOTAL_SALES
   FROM sales2
   GROUP BY YEAR(ORDERDATE), MONTH(ORDERDATE)
   ORDER BY YEAR(ORDERDATE), MONTH(ORDERDATE);

10.SELECT PRODUCTLINE, AVG(PRICEEACH) AS average_price
   FROM sales2
   GROUP BY PRODUCTLINE
   HAVING AVG(PRICEEACH) > (SELECT AVG(PRICEEACH) FROM sales2);
